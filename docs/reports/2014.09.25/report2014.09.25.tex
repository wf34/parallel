\section{Analysis}
\begin{align*}
    &  n = \text{problem size}\\
    &  p = \text{processors amont}\\
    &  s = \text{chunk stride} = \alpha  p\\
    &  l = \text{chunk length} = n/s\\
    &  T_{\text{comp}} = (2s-1) \cdot {\frac{n^2}{s^2}} = O(\frac{n^2}{p})\\
    &  T_{\text{comm}} = (2s-1) \cdot l = O(n)\\
    &  T_{\text{sync}} = 2s-1 = O(p)
\end{align*}
